import tkinter as tk
from PIL import ImageTk, Image
import re

def update_preview():
    # Obtém o código ZPL do campo de texto
    zpl_code = text_input.get("1.0", tk.END)

    # Analisa o código ZPL para extrair informações sobre os elementos de texto
    text_elements = []
    for match in re.finditer(r'\^FO(\d+),(\d+)\^A[^\^]+\^FD([^\^]+)\^FS', zpl_code):
        x, y, text = match.groups()
        text_elements.append((int(x), int(y), text))
        print(f"Encontrado elemento de texto: x={x}, y={y}, text={text}")

    # Limpa o canvas
    canvas.delete("all")

    # Adiciona um retângulo para delimitar a área de visualização da etiqueta
    canvas.create_rectangle(10, 10, 290, 290)

    # Cria objetos gráficos interativos para representar os elementos de texto
    for x, y, text in text_elements:
        item = canvas.create_text(x, y, text=text, anchor=tk.NW)
        canvas.tag_bind(item, "<ButtonPress-1>", on_text_press)
        canvas.tag_bind(item, "<B1-Motion>", on_text_drag)


def on_text_press(event):
    # Armazena a posição atual do elemento de texto
    canvas.current_text = canvas.find_withtag("current")
    canvas.current_x = event.x
    canvas.current_y = event.y

def on_text_drag(event):
    # Move o elemento de texto para a nova posição
    dx = event.x - canvas.current_x
    dy = event.y - canvas.current_y
    canvas.move(canvas.current_text, dx, dy)
    canvas.current_x = event.x
    canvas.current_y = event.y

def apply_changes():
    # Obtém o código ZPL do campo de texto
    zpl_code = text_input.get("1.0", tk.END)

    # Atualiza as coordenadas dos elementos no código ZPL
    for item in canvas.find_all():
        if canvas.type(item) == "text":
            x, y = canvas.coords(item)
            text = canvas.itemcget(item, "text")
            zpl_code = re.sub(rf'(\^FO)\d+,\d+(\^A[^\^]+\^FD{text}\^FS)', rf'\g<1>{int(x)},{int(y)}\g<2>', zpl_code)

    # Atualiza o campo de texto com o novo código ZPL
    text_input.delete("1.0", tk.END)
    text_input.insert(tk.END, zpl_code)

# Cria a janela principal
root = tk.Tk()
root.title("Visualizador de Etiquetas ZPL")

# Cria o campo de texto para entrada de código ZPL
text_input = tk.Text(root, height=10, width=30)
text_input.pack()

# Cria o botão para atualizar a visualização da etiqueta
update_button = tk.Button(root, text="Atualizar", command=update_preview)
update_button.pack()

# Cria o botão para aplicar as mudanças no código ZPL
apply_button = tk.Button(root, text="Aplicar", command=apply_changes)
apply_button.pack()

# Cria o canvas para exibir a imagem da etiqueta
canvas = tk.Canvas(root, width=300, height=300)
canvas.pack()

# Inicia o loop principal da janela
root.mainloop()
